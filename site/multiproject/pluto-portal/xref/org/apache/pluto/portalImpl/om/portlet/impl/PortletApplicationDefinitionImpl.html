124     private String getGUID()
125     {
126         if (GUID == null) {
127             GUID = "";            
128             String id = "";
129 
130             if (webApplication != null) {
131                 id = webApplication.getContextRoot();
132             } else {
133                 id = contextPath;
134             }
135 
136             if (id!=null) {
137                 if (id.startsWith("/")) {
138                     id = id.substring(id.indexOf("/")+1);
139                 }
140 
141                 GUID += id;
142             }
143         }
144 
145         return GUID;
146     }

231     public String toString(int indent)
232     {
233         StringBuffer buffer = new StringBuffer(50);
234         StringUtils.newLine(buffer,indent);
235         buffer.append(getClass().toString()); buffer.append(":");
236         StringUtils.newLine(buffer,indent);
237         buffer.append("{");
238         StringUtils.newLine(buffer,indent);
239         buffer.append("objectID='"); buffer.append(getId().toString()); buffer.append("'");
240         StringUtils.newLine(buffer,indent);
241         buffer.append("GUID='"); buffer.append(GUID); buffer.append("'");
242         StringUtils.newLine(buffer,indent);
243         buffer.append("version='"); buffer.append(version); buffer.append("'");
244 
245         Iterator iterator = portlets.iterator();
246         if (iterator.hasNext()) {
247             StringUtils.newLine(buffer,indent);
248             buffer.append("Portlets:");
249         }
250         while (iterator.hasNext()) {
251             buffer.append(((PortletDefinitionImpl)iterator.next()).toString(indent+2));
252         }
253         if (webApplication!=null) {
254             StringUtils.newLine(buffer,indent);
255             buffer.append("webApplication:");
256             buffer.append(((WebApplicationDefinitionImpl)webApplication).toString(indent+2));
257         }
258         StringUtils.newLine(buffer,indent);
259         buffer.append("}");
260         return buffer.toString();
261     }
